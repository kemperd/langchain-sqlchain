-- CHAINS definition

CREATE COLUMN TABLE "CHAINS" (
	"Chain" NVARCHAR(50),
	"DESCRIPTION" NVARCHAR(50)
);

-- CITIES definition

CREATE COLUMN TABLE "CITIES" (
	"CITY" NVARCHAR(50),
	"DESCRIPTION" NVARCHAR(50),
	"STATE" INTEGER
);

-- PRODUCT_GROUPS definition

CREATE COLUMN TABLE "PRODUCT_GROUPS" (
	"PRODUCT_GROUP" NVARCHAR(50),
	"DESCRIPTION" NVARCHAR(50)
);

-- PRODUCTS definition

CREATE COLUMN TABLE "PRODUCTS" (
	"PRODUCT" NVARCHAR(50),
	"DESCRIPTION" NVARCHAR(50),
	"PRODUCT_GROUP" NVARCHAR(50)
);

-- REGIONS definition

CREATE COLUMN TABLE "REGIONS" (
	"REGION" INTEGER,
	"DESCRIPTION" NVARCHAR(50)
);

-- STATES definition

CREATE COLUMN TABLE "STATES" (
	"STATE" INTEGER,
	"DESCRIPTION" NVARCHAR(50),
	"REGION" INTEGER
);

-- STORE_BRANDS definition

CREATE COLUMN TABLE "STORE_BRANDS" (
	"STORE_BRAND" NVARCHAR(50),
	"DESCRIPTION" NVARCHAR(50),
	"Chain" NVARCHAR(50)
);

-- STORES definition

CREATE COLUMN TABLE "STORES" (
	"STORE" NVARCHAR(50),
	"DESCRIPTION" NVARCHAR(50),
	"STORE_BRAND" NVARCHAR(50)
);

-- TRANSACTIONS definition

CREATE COLUMN TABLE "TRANSACTIONS" (
	"VERSION" NVARCHAR(50),
	"ORDER_ID" INTEGER,
	"Date" DATE,
	"STORE" NVARCHAR(50),
	"STORE_ID" NVARCHAR(50),
	"LONGITUDE" NVARCHAR(50),
	"LATITUDE" NVARCHAR(50),
	"CITY_ID" NVARCHAR(50),
	"CITY" NVARCHAR(50),
	"PRODUCT_ID" NVARCHAR(50),
	"PRICE" DECIMAL(10,2),
	"QUANTITY_SOLD" INTEGER,
	"ORIGINAL_SALES_PRICE" DECIMAL(10,2),
	"DISCOUNT_PERCENT" DECIMAL(10,2),
	"DISCOUNT" DECIMAL(10,2),
	"SALES_REVENUE" DECIMAL(10,2),
	"GROSS_MARGIN" DECIMAL(10,2)
);
